    function getUsuarios() {

        $this->db->from('usuarios');
        $this->db->limit('100');
        $query = $this->db->get();
        if ($query) {
            $result = $query->result();
            return $result;
        } else {
            $result = false;
        }
    }

    function getDados($id) {

        $this->db->where('idUsuarios', $id);
        $this->db->limit(1);
        return $this->db->get('usuarios')->row();
    }

    public function check_email($email) {
        $this->db->where('email', $email);
